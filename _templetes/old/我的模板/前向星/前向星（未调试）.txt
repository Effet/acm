//有向图链式前向星即是数组模拟链表，需先初始化head和next
typedef struct forward_table{int to, next;}FwdT;
int n, m, head[MAX_VETX];
FwdT F[MAX_EDGE];
//第i条从u到v的边
void insertDirectEdge(int i, int u, int v){
   F[i].to = v, F[i].next = head[u], head[u] = i;
}
