方法一：采用字符串函数，按空格判断

#include "string.h"
#include "stdio.h"
void main() 
{
 char a[]="She is my sister !"; 
 printf("%s,strlen(a) = %d\n",&a,strlen(a));
 int i,j,k=0; 
 for(i=strlen(a);i>=0;i--) 
 { 
  k++; 
  //printf("k=%d,i=%d,a[i]=%c\n",k,i,a[i]);
  if(a[i]!=' '&&i>0)
   continue; 
  for(j=i;j<k+i;j++) 
   printf("%c",a[j]); 
  printf(" ");
  k=0; 
 } 
 printf("\n");
}

方法二：使用系统函数reverse，不过这个只是全部转置
#include<iostream.h>
#include<string.h>

void reverse(char*str,int n){
	if (n<=1) return;
	char temp=str[n-1];
	str[n-1]=str[0];
	str[0]=temp;
	reverse(str+1,n-2);
}

void main()
{
 char str[100];
 cout<<"please input a string:"<<endl;
 cin.get(str,100);
 reverse(str,strlen(str));

 cout<<"after reverse:"<<endl<<str<<endl;
}

int numofwords(char inputs[])
{
 int a,number=0;
 for (a=1;a<=MAXLETTER-1;a++)
 {
  if ((inputs[a]==' '||inputs[a]=='\0')&&inputs[a-1]!=' ')
   number++;
  if (inputs[a]=='\0')
   break;
 }
 return number;
}

void storage(char inputs[])
{
 int a=0,b;
 position=0;
 while(position<=MAXLETTER)
 {
  if (inputs[position]=='\0')
   break;
  b=0;
  while (inputs[position]!=' ')
  {
   words[a][b]=inputs[position];
   words[a][b+1]='\0';
   position++;
   b++;
   if (inputs[position]==' ')
    a++;
  }
  position++;
 } 
}

#include "string.h"
#include "stdio.h"

char * Reverse(char *str);
void main() 
{
 char a[]="She is my sister !"; 
 printf("%s,strlen(a) = %d\n",&a,strlen(a));

 char * b = Reverse(a);
 printf("Reverse string:\n\t%s\n",b);
}

char * Reverse(char *str)
{
 char *b = new char[sizeof(str)+1];
 memset(b,sizeof(str)+1,0);
 char *p;
 p = b;

 int i,j,k=0; 
 for(i=strlen(str)-1;i>=0;i--) 
 { 
  k++; 
  if(str[i]!=' ' && i > 0)
   continue;

  for(j=i;j<k+i;j++) 
  {
   *p++ = str[j];
   //printf("str[j] = %c,j=%d\n",str[j],j);
  }
  //printf("b = %s\n",b);
  k=0; 
 } 
 *p = '\0';

 return b;
}

#include "string.h"
#include "stdio.h"

char * my_strrev(char *str);
void main() 
{
 char a[]="She is my sister !"; 
 printf("%s,strlen(a) = %d\n",&a,strlen(a));
 
 char * b = my_strrev(a);
 printf("Reverse sting:\n\t%s\n",b);
}
char * my_strrev(char *str)
{
 char *right = str;
 char *left = str;
 char ch;

 while (*right) 
  right++;
 right--;
 while (left < right)
 {
  ch = *left;
  *left++ = *right;
  *right-- = ch;
 }
 return(str);
}
