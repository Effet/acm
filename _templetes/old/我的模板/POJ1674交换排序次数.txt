#include <stdio.h>
#include <string.h>
#define MAXN 10000+10

int a[MAXN];
int b[MAXN];

void circ(int n, int *a, int *b, int i){
	if(b[i]) return;
	b[i] = 1;
	circ(n, a, b, a[i]);
}


int solv(int n, int *a){
	int i, ans = n;
	memset(b, 0, sizeof(b));
	for(i=1; i<=n; i++){
		if(!b[i]){
			circ(n, a, b, i);
			ans--;
		}
	}
	return ans;
}

int main(){
	int t, n, i;
	while(scanf("%d", &t) != EOF){
		while(t--){
			scanf("%d", &n);
			for(i=1; i<=n; i++)
				scanf("%d", &a[i]);
			printf("%d\n", solv(n, a));
		}
	}
	return 0;
}
